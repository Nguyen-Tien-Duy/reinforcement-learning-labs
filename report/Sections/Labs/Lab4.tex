\section{Mục tiêu bài Lab 4}
Lab 4 tập trung vào việc triển khai và tối ưu hóa thuật toán Deep Q-Network (DQN) - một cột mốc quan trọng trong Deep Reinforcement Learning:
\begin{itemize}
    \item Hiểu cách sử dụng Neural Network để xấp xỉ hàm Q (Function Approximation).
    \item Triển khai cơ chế Experience Replay để xóa bỏ sự tương quan giữa các mẫu dữ liệu.
    \item Sử dụng Target Network để ổn định quá trình tối ưu hóa mục tiêu Bellman.
    \item Nắm vững các kỹ thuật ổn định training như Soft Update và Double DQN.
\end{itemize}

\begin{lstlisting}[style=PythonStyle, caption={Soft Update \& DQN Learn}]
def learn(self, experiences, gamma):
    states, actions, rewards, next_states, dones = experiences
    # Double DQN action selection & evaluation
    best_action = self.qnetwork_local(next_states).argmax(1).unsqueeze(1)
    Q_targets_next = self.qnetwork_target(next_states).gather(1, best_action)
    # Compute targets and expected values
    Q_targets = rewards + (gamma * Q_targets_next * (1 - dones))
    Q_expected = self.qnetwork_local(states).gather(1, actions)
    # Loss & Optimization
    loss = F.mse_loss(Q_expected, Q_targets)
    self.optimizer.zero_grad(); loss.backward(); self.optimizer.step()
    # Update target network
    self.soft_update(self.qnetwork_local, self.qnetwork_target, TAU)

def soft_update(self, local_model, target_model, tau):
    # theta_target = tau * theta_local + (1 - tau) * theta_target
    for target_param, local_param in zip(target_model.parameters(), local_model.parameters()):
        target_param.data.copy_(tau * local_param.data + (1.0 - tau) * target_param.data)
\end{lstlisting}
\section{Thiết lập và Cài đặt}

\subsection{4.1 Q-function Approximation với Neural Network}
Thay vì lưu trữ Q-table, chúng ta sử dụng một mạng Neural (MLP) với kiến trúc:
\begin{itemize}
    \item \textbf{Input:} Vector trạng thái 4 chiều (CartPole-v1).
    \item \textbf{Hidden Layers:} 2 lớp ẩn, mỗi lớp 128 units, sử dụng hàm kích hoạt ReLU.
    \item \textbf{Output:} 2 nodes tương ứng với giá trị Q của các hành động (Trái/Phải).
\end{itemize}

\textbf{Trả lời câu hỏi lý thuyết:}
\begin{enumerate}
    \item \textbf{Tại sao Q-table không dùng được cho state liên tục?} Vì số lượng trạng thái là vô hạn, không thể biểu diễn bằng một bảng hữu hạn. Ngoài ra, Q-table không thể "tổng quát hóa" kiến thức từ trạng thái này sang trạng thái lân cận.
    \item \textbf{Function Approximation là gì?} Là việc sử dụng một hàm số có tham số (như mạng Neural) để xấp xỉ giá trị của một hàm số phức tạp (hàm Q). Thay vì lưu từng giá trị đơn lẻ, ta chỉ cần lưu bộ trọng thố $\theta$.
    \item \textbf{Tại sao Output layer = số action?} Để trong một lần forward pass, mạng có thể dự đoán đồng thời giá trị Q cho tất cả các hành động khả thi, giúp việc chọn hành động ($\text{argmax}$) hiệu quả hơn.
    \item \textbf{Vai trò của ReLU?} ReLU giúp mạng học được các mối quan hệ phi tuyến tính phức tạp giữa trạng thái và giá trị Q, điều mà các hàm tuyến tính không làm được.
    \item \textbf{Neural Network giúp Generalize như thế nào?} Các trạng thái gần nhau trong không gian vector sẽ kích hoạt các neuron tương tự nhau, dẫn đến đầu ra Q-value tương tự. Do đó, Agent có thể đưa ra quyết định hợp lý cho cả những trạng thái nó chưa từng gặp trực tiếp.
\end{enumerate}

\subsection{4.2 Deep Q-Learning và Target Network}
Thuật toán DQN cập nhật trọng số dựa trên cực tiểu hóa hàm Loss:
\begin{equation}
    L = \mathbb{E} \left[ (r + \gamma \max_{a'} Q_{target}(s', a') - Q_{local}(s, a))^2 \right]
\end{equation}

\textbf{Trả lời câu hỏi lý thuyết:}
\begin{enumerate}
    \item \textbf{Tại sao target dùng max Q(s', a)?} Dựa trên phương trình Bellman tối ưu, khẳng định rằng giá trị của trạng thái hiện tại bằng phần thưởng tức thời cộng với giá trị kỳ vọng tốt nhất từ trạng thái kế tiếp.
    \item \textbf{Bootstrapping trong DQN là gì?} Là việc sử dụng dự đoán của chính mạng Neural ở tương lai ($Q(s', a')$) để cập nhật giá trị hiện tại.
    \item \textbf{Nếu target thay đổi liên tục?} Quá trình học sẽ cực kỳ mất ổn định (hiện tượng "đuổi theo mục tiêu di động"). Target Network giúp cố định mục tiêu này trong một khoảng thời gian để mạng Local có thể hội tụ.
    \item \textbf{So sánh với Q-learning (tabular):} DQN có khả năng xử lý không gian trạng thái khổng lồ và khả năng tổng quát hóa, trong khi Tabular Q-learning chỉ hoạt động tốt trên các môi trường rất nhỏ và rời rạc.
\end{enumerate}

\subsection{4.3 Experience Replay}
Cơ chế Experience Replay lưu trữ bộ $(s, a, r, s', done)$ vào một Buffer và lấy mẫu ngẫu nhiên (Mini-batch) để huấn luyện.

\begin{lstlisting}[style=PythonStyle, caption={Experience Replay Implementation}]
class ReplayMemory:
    def __init__(self, action_size, buffer_size, batch_size, device):
        self.memory = deque(maxlen=buffer_size)
        self.batch_size = batch_size
    def add(self, state, action, reward, next_state, done):
        self.memory.append((state, action, reward, next_state, done))
    def sample(self):
        # Shuffling to break temporal correlation
        experiences = random.sample(self.memory, k=self.batch_size)
        # Convert to Tensors...
        return (states, actions, rewards, next_states, dones)
\end{lstlisting}
\textbf{Trả lời câu hỏi lý thuyết:}
\begin{enumerate}
    \item \textbf{Vi phạm giả định i.i.d:} Các thuật toán tối ưu hóa dựa trên Gradient (như Adam, SGD) đều dựa trên giả định ngầm rằng dữ liệu huấn luyện là \textbf{độc lập và phân phối chuẩn (i.i.d)}. Trong RL, các trải nghiệm kế tiếp trong một episode có sự tương quan thời gian (temporal correlation) cực cao ($s_{t+1}$ phụ thuộc vào $s_t$).
    \item \textbf{Hậu quả của sự tương quan:} Nếu huấn luyện trực tiếp theo chuỗi, Gradient của mỗi mini-batch sẽ bị chệch (biased) theo quỹ đạo hiện tại, dẫn đến việc mạng Neural bị "học vẹt" con đường vừa đi qua và xảy ra hiện tượng \textbf{Quên thảm họa (Catastrophic Forgetting)} đối với các kinh nghiệm cũ.
    \item \textbf{Cơ chế phá vỡ tương quan của Replay Buffer:} Bằng cách lưu trữ và lấy mẫu ngẫu nhiên (shuffling), Replay Buffer biến dữ liệu chuỗi thời gian thành một tập hợp "có vẻ" độc lập, giúp ước lượng Gradient trở nên không chệch và ổn định quá trình hội tụ.
    \item \textbf{Đánh đổi về kích thước Buffer:} Buffer quá nhỏ sẽ không đủ sức phá vỡ tương quan. Buffer quá lớn có thể chứa quá nhiều dữ liệu "lỗi thời" từ các chính sách cũ (Stale data), làm chậm tốc độ cải thiện của chính sách hiện tại.
\end{enumerate}

\textbf{Nhận xét đồ thị (Hình \ref{fig:lab4_reward}):}
\begin{itemize}
    \item \textbf{Tốc độ hội tụ:} Agent đạt mức Reward tối đa 500/500 điểm chỉ sau khoảng 400-600 episodes. Đây là kết quả vượt trội so với DQN thông thường ở Lab 3, nơi Reward dao động liên tục và không ổn định xuyên suốt quá trình huấn luyện.

    \item \textbf{Tại sao ổn định hơn Lab 3? — Vai trò của Double DQN:}
    DQN thông thường dùng cùng một mạng để vừa \textit{chọn hành động} vừa \textit{định giá trị}, dẫn đến Overestimation Bias (đánh giá quá cao các Q-value nhiễu). Double DQN tách hai nhiệm vụ này: mạng \textbf{Local} chọn hành động ($\arg\max$), còn mạng \textbf{Target} định giá trị cho hành động đó. Sự phân tách này triệt tiêu phần lớn Overestimation Bias, khiến các đột biến Loss và Reward giảm đi đáng kể.

    \item \textbf{Tại sao ổn định hơn Lab 3? — Vai trò của Soft Update ($\tau = 0.001$):}
    Thay vì sao chép nguyên xi trọng số từ mạng Local sang mạng Target sau mỗi $C$ bước (Hard Update), Soft Update chỉ dịch chuyển mạng Target một lượng rất nhỏ ($\theta_{target} \leftarrow \tau \theta_{local} + (1-\tau)\theta_{target}$). Điều này loại bỏ hoàn toàn hiện tượng mục tiêu nhảy vọt đột ngột (moving target problem), khiến Loss giảm đều đặn hơn thay vì bị các đột biến bất thường như Lab 3.

    \item \textbf{Giai đoạn ổn định (Plateau):} Sau khi đạt 500 điểm, đường Reward duy trì ổn định ở mức tối đa mà không bị tụt xuống — đây là bằng chứng cho thấy Policy đã hội tụ thực sự chứ không chỉ là may mắn ngẫu nhiên.
\end{itemize}

\textbf{Trả lời câu hỏi phân tích:}
\begin{enumerate}
    \item \textbf{Khi nào model học tốt (loss)?} Khi hàm Loss giảm dần và dao động quanh một giá trị thấp, đồng thời xu hướng chung của Reward là tăng lên. Lưu ý: Loss trong RL không phản ánh trực tiếp chất lượng Policy như trong Supervised Learning.
    \item \textbf{Khi nào agent đạt policy tốt (reward)?} Khi đường Reward trung bình tiệm cận mức trần của môi trường (500 điểm) và duy trì ổn định qua nhiều episodes.
    \item \textbf{Sự bất ổn định và "The Deadly Triad":} Deep RL khó hội tụ hơn Tabular RL do sự kết hợp của 3 yếu tố (Tam giác chết): (1) \textbf{Function Approximation} (thay đổi một trọng số ảnh hưởng toàn bộ không gian trạng thái); (2) \textbf{Bootstrapping} (dùng dự đoán để học dự đoán); và (3) \textbf{Off-policy training} (dữ liệu học khác với chính sách hiện tại). Sự kết hợp này dễ dẫn đến việc Q-value bị khuếch đại sai số và phân kỳ.
    \item \textbf{Bản chất toán học của Overestimation Bias:} Toán tử $\max$ trong mục tiêu Bellman luôn chọn giá trị lớn nhất, điều này có xu hướng chọn cả những sai số dương (positive noise) trong ước lượng Q-value. Hệ quả là mạng DQN thường đánh giá quá cao giá trị của các hành động. Chúng ta đã khắc phục điều này bằng \textbf{Double DQN}: dùng mạng Online để chọn hành động ($\text{argmax}$) and mạng Target để định giá, giúp tách biệt việc chọn lọc và định giá, từ đó triệt tiêu sai số hệ thống này.
\end{enumerate}
